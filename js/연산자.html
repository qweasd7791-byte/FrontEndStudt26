// 기본 산술 연산자들 let a = 10; let b = 3; console.log("덧셈:", a + b); // 13 console.log("뺄셈:", a - b); // 7 console.log("곱셈:", a * b); // 30 console.log("나눗셈:", a / b); // 3.333... console.log("나머지:", a % b); // 1 (10을 3으로 나눈 나머지) console.log("거듭제곱:", a ** 2); // 100 (10의 2제곱) // 문자열과 숫자의 덧셈 (주의!) let name = "김개발"; let age = 20; console.log(name + age); // "김개발20" (문자열 연결) console.log(age + 5); // 25 (숫자 계산)
